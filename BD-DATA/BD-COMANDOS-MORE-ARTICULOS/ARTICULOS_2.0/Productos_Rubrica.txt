CREATE DATABASE productosRubrica;

USE productosRubrica;

-- LISTA

CREATE TABLE Cliente(
    Id_Cliente int(5) PRIMARY KEY, 
    Id_Direccion int(5), 

    Nombre varchar(20), 
    Apellido varchar(20), 
    Saldo double, 
    Lim_credito double, 
    Porcentage_Descuento int(2)
);

-- LISTA

CREATE TABLE Direccion_Cliente(
    Id_Direccion int(5) PRIMARY KEY, 

    Estado varchar(20),
    Colonia varchar(20),
    Calle varchar(20),
    CP int(5)
);

-- LISTA

CREATE TABLE Cabecera(
    Id_Cabecera int(5) PRIMARY KEY,
    Id_Cliente int(5), 
    Id_Direccion int(5),

    Fecha_Compra date
);

-- LISTA
    
CREATE TABLE Cuerpo(
    Id_Cuerpo int(5) PRIMARY KEY,
    Id_Articulo int(5),

    Articulos_vendidos int(2)
);

-- LISTA

CREATE TABLE Articulos(
    Id_Articulo int(5) PRIMARY KEY,
    Id_Fabrica int(5),

    Articulo varchar(50),
    Tipo varchar(20),
    Precio float
);

-- LISTA

CREATE TABLE Fabrica(
    Id_Fabrica int(5) PRIMARY KEY,

    Nombre varchar(20),
    Estado varchar(20),
    Telefono bigint(10)
);


-- LISTA

CREATE TABLE Pedido(
    Id_Pedido  int(5) PRIMARY KEY,
    Id_Cliente int (5), 
    Id_Articulo int(5),
    Id_Cabecera int(5), 
    Id_Cuerpo int(5)
);

-- indices de las tablas

CREATE INDEX Direccion ON cliente(Id_Direccion);

CREATE INDEX Cliente ON Cabecera(Id_Cliente);
CREATE INDEX Direccion2 ON Cabecera(Id_Direccion);

CREATE INDEX pedidoArticulo ON Cuerpo(Id_Articulo);

CREATE INDEX Fabrica ON Articulos(Id_Fabrica);

CREATE INDEX Id_Cliente ON Pedido(Id_Cliente);
CREATE INDEX Id_Articulo ON Pedido(Id_Articulo);
CREATE INDEX Id_Cabecera ON Pedido(Id_Cabecera);
CREATE INDEX Id_Cuerpo ON Pedido(Id_Cuerpo);

-- relaciones

ON DELETE CASCADE ON UPDATE CASCADE;

ALTER TABLE Cabecera ADD FOREIGN KEY (Id_Cliente) REFERENCES cliente(Id_Cliente) 
ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE Cabecera ADD FOREIGN KEY (Id_Direccion) REFERENCES Direccion_Cliente(Id_Direccion) 
ON DELETE CASCADE ON UPDATE CASCADE;

ALTER TABLE Articulos ADD FOREIGN KEY (Id_Fabrica) REFERENCES Fabrica(Id_Fabrica) 
ON DELETE CASCADE ON UPDATE CASCADE;

ALTER TABLE Cuerpo ADD FOREIGN KEY (Id_Articulo) REFERENCES Articulos(Id_Articulo) 
ON DELETE CASCADE ON UPDATE CASCADE;

ALTER TABLE Pedido ADD FOREIGN KEY (Id_Cliente) REFERENCES Cliente(Id_Cliente) 
ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE Pedido ADD FOREIGN KEY (Id_Articulo) REFERENCES Articulos(Id_Articulo) 
ON DELETE CASCADE ON UPDATE CASCADE;

ON DELETE CASCADE ON UPDATE CASCADE;
