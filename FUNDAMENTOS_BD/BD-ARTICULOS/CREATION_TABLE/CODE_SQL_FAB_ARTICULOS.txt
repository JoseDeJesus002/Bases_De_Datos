TABLA CREACION

CREATE TABLE ARTICULOS
(Numero_Pedido INT(10) PRIMARY KEY,Id_Cuerpo INT(10),Descripcion VARCHAR(50));

CREATE TABLE ARTICULOS_FABRICA
(Numero_Pedido INT(10),Id_Fabrica INT(10),Descripcion VARCHAR(50));

CREATE TABLE CABECERA
(Id_Cabecera INT(10) PRIMARY KEY,Direccion TEXT,Fecha_Pedido DATE);

CREATE TABLE CLIENTE
(Id_Cliente INT(10) PRIMARY KEY,Direccion TEXT,Limite_Credito INT(5),Saldo DOUBLE,Descuento INT(2),Nombre VARCHAR(50));

CREATE TABLE CUERPO
(Id_Cuerpo INT(10) PRIMARY KEY,Cantidad INT(6));

CREATE TABLE FABRICA
(Id_Fabrica INT(10) PRIMARY KEY,Telefono BIGINT(13));

CREATE TABLE PEDIDOS
(Id_Pedidos INT(10) PRIMARY KEY,Id_Cabecera INT(10),Id_Cuerpo INT(10),Id_Cliente INT(10));
