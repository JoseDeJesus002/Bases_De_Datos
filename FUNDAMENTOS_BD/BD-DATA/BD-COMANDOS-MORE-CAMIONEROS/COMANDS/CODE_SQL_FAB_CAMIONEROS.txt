CREATE TABLE CAMIONEROS
(CURP VARCHAR(18) PRIMARY KEY,Matricula VARCHAR(8),Id_Paquete INT(10),Nombre VARCHAR(50),Telefono BIGINT(13),Direccion TEXT, Poblacion INT(10),Salario DOUBLE);

CREATE TABLE CAMIONES
(Matricula VARCHAR(8) PRIMARY KEY,Modelo TEXT,Tipo TEXT,Potencia DOUBLE);

CREATE TABLE PAQUETE
(Id_Paquete INT(10) PRIMARY KEY,Id_Provincia INT(10),Descripcion TEXT,Destinatario TEXT,Direccion_Destinatario TEXT);

CREATE TABLE PROVINCIA
(Id_Provincia INT(10) PRIMARY KEY,Nombre TEXT);

INDICES

CAMIONEROS
CREATE UNIQUE INDEX Matricula ON CAMIONEROS(Matricula);
CREATE UNIQUE INDEX Id_Paquete ON CAMIONEROS(Id_Paquete);

PAQUETE
CREATE UNIQUE INDEX Id_Provincia ON PAQUETE(Id_Provincia);


RELACIONES

CAMIONEROS
ALTER TABLE CAMIONEROS ADD FOREIGN KEY (Matricula) REFERENCES CAMIONES (Matricula) ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE CAMIONEROS ADD FOREIGN KEY (Id_Paquete) REFERENCES PAQUETE (Id_Paquete) ON DELETE CASCADE ON UPDATE CASCADE;

PAQUETE
ALTER TABLE PAQUETE ADD FOREIGN KEY (Id_Provincia) REFERENCES PROVINCIA (Id_Provincia) ON DELETE CASCADE ON UPDATE CASCADE;
