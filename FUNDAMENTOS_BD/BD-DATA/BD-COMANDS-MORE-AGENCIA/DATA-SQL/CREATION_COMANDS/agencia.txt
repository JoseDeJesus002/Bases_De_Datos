creacion de tabla

CREATE TABLE AGENCIA
(ID_AGENCIA INT(5) PRIMARY KEY,ID_VUELO INT(5),ID_HOTEL INT(5),ID_SUCURSAL INT(5));

CREATE TABLE VUELO
(ID_VUELO INT(5) PRIMARY KEY,ID_CLASE INT(5),FH_SALIDA  DATETIME,FH_LLEGADA  DATETIME,ORIGEN TEXT,DESTINO TEXT, NO_PLAZAS INT(5));

CREATE TABLE CLASE
(ID_CLASE INT(5) PRIMARY KEY,TIPO TEXT);

CREATE TABLA CLIENTE
(CURP VARCHAR(20) PRIMARY KEY,ID_SUCURSAL INT(5),ID_AGENCIA INT(5),NOMBRE TEXT,APELLIDO TEXT,TELEFONO BIGINT(13),E_MAIL VARCHAR(20));

CREATE TABLE HOTEL
(ID_HOTEL INT(5) PRIMARY KEY,ID_REGIMEN INT(5),NO_ESTRELLAS INT(5),FH_SALIDA  DATETIME,FH_ENTRADA  DATETIME, NOMBRE TEXT,DIRECCION TEXT);

CREATE TABLE SUCURSAL
(ID_SUCURSAL INT(5) PRIMARY KEY,ID_AGENCIA INT(5),DIRECCION TEXT,PROVINCIA TEXT, CD TEXT,TELEFONO BIGINT(13));

CREATE TABLE REGIMEN
(ID_REGIMEN INT(5) PRIMARY KEY);
